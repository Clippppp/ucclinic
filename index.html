📋 구글 시트 연동 방법 (클릭하여 펼치기)
  1. 구글 드라이브에서 새 Google 스프레드시트를 만듭니다.
  2. 상단 메뉴 → 확장 프로그램 → Apps Script를 클릭합니다.
  3. 아래 코드를 붙여넣고 저장(Ctrl+S)합니다:
function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var data  = JSON.parse(e.postData.contents);

  if (sheet.getLastRow() === 0) {
    sheet.appendRow(['ID','이름','학교','학년','연락처','신청시간대','클리닉종류','세부내용','추가요청사항','제출시각']);
  }
  sheet.appendRow([
    data.id, data.name, data.school, data.grade, data.contact,
    data.timeSlot, data.subject, data.detail, data.note, data.submittedAt
  ]);
  return json({ result: 'success' });
}

function doGet(e) {
  var props  = PropertiesService.getScriptProperties();
  var stored = props.getProperty('ADMIN_PW');
  var action = e.parameter.action;

  // ── 신청 목록 반환 ──
  if (action === 'getSubmissions') {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    var rows  = sheet.getDataRange().getValues();
    if (rows.length <= 1) return json({ data: [] });
    var keys = ['id','name','school','grade','contact','timeSlot','subject','detail','note','submittedAt'];
    var result = rows.slice(1).map(function(row) {
      var obj = {};
      keys.forEach(function(k, i) { obj[k] = row[i] ? row[i].toString() : ''; });
      return obj;
    });
    return json({ data: result });
  }

  // ── 행 삭제 ──
  if (action === 'deleteEntry') {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    var id    = e.parameter.id;
    var rows  = sheet.getDataRange().getValues();
    for (var i = 1; i < rows.length; i++) {
      if (rows[i][0].toString() === id) {
        sheet.deleteRow(i + 1);
        return json({ ok: true });
      }
    }
    return json({ ok: false, msg: '해당 항목을 찾을 수 없습니다.' });
  }

  // ── 비밀번호 관련 ──
  if (action === 'checkPw')  return json({ hasPw: !!stored });
  if (action === 'verifyPw') return json({ ok: !stored || e.parameter.pw === stored });
  if (action === 'setPw') {
    if (stored) return json({ ok: false, msg: '이미 비밀번호가 설정되어 있습니다.' });
    props.setProperty('ADMIN_PW', e.parameter.pw);
    return json({ ok: true });
  }
  if (action === 'changePw') {
    if (e.parameter.current !== stored) return json({ ok: false, msg: '현재 비밀번호가 올바르지 않습니다.' });
    props.setProperty('ADMIN_PW', e.parameter.next);
    return json({ ok: true });
  }

  return json({ status: 'ok' });
}

function json(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

🔗 학생 공유 링크 만들기 — 무료 배포 방법 (클릭하여 펼치기)

아래 두 가지 방법 중 하나로 HTML 파일을 배포하면 학생들에게 나눌 수 있는 고정 URL이 생깁니다. 두 방법 모두 완전 무료입니다.

✦ 추천 · 가장 쉬운 방법
방법 1 — Netlify Drop (드래그 앤 드롭)
1
app.netlify.com/drop 에 접속합니다.
2
다운로드한 clinic-signup.html 파일을 화면에 드래그 앤 드롭합니다.
3
자동으로 배포되며 https://랜덤.netlify.app 형식의 URL이 생성됩니다.
4
해당 URL을 카카오톡, 문자 등으로 학생들에게 공유합니다. (무료 계정 생성 시 URL 커스텀도 가능)
GitHub Pages
방법 2 — GitHub Pages
1
github.com 에서 무료 계정을 만들고 New Repository를 클릭합니다.
2
파일명을 index.html 로 바꾼 뒤 업로드합니다.
3
저장소 설정 → Pages → Source: main branch 로 설정합니다.
4
https://사용자명.github.io/저장소명 주소로 접속 가능해집니다.

💡 내용 수정 후에는 파일을 다시 업로드(재배포)하면 반영됩니다.
